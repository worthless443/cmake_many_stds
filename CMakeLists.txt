# Example 
# Compile test/test.cc with most recent version of std for C++
cmake_minimum_required(VERSION 3.14)
project(BENCHMARK_STANDARDS CXX)
include(build_standards.cmake)
detect_gcc_13(has_gcc_13 current_is_gcc13)
if(has_gcc_13 AND NOT current_is_gcc13)
	message(STATUS "The all new GCC-13 has been found on your system, but you are not using it")
elseif(current_is_gcc13)
	message(WARNING "if you are using standard C++ headers, there might be problem")
endif()
add_executable_with_all_std_versions(6 test tests/test.cc)
add_benchmark(test)
add_executable_with_all_std_versions(6 crazy tests/crazy.cc)
add_benchmark(crazy)
